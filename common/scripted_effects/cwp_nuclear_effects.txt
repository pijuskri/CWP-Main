effect_reset_relevant_coverage = {
	clear_variable_list = strategic_regions
	every_state_region = {
		limit = {
			is_state_region_land = yes
		}
		root = {
			if = {
				limit = {
					PREV = {
						any_scope_state = {
							ROOT = {
								is_target_in_variable_list = {
									name = wmd_target_list
									target = PREV
								}
							}
						}
					}
				}
				add_to_variable_list = {
					name = strategic_regions
					target = PREV.region
				}
			}
		}
	}
}
