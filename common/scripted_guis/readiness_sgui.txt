raise_readiness = {
	effect = {
		change_variable = {
			name = var_readiness
			add = 1
		}
		set_variable = {
			name = changed_readiness
			days = 30
		}
	}
}

lower_readiness = {
	effect = {
		change_variable = {
			name = var_readiness
			subtract = 1
		}
		set_variable = {
			name = changed_readiness
			days = 30
		}
	}
}

can_lower_readiness = {
	is_shown = {
		var:var_readiness > 1
		not = {
			has_variable = changed_readiness
		}
	}
}

can_raise_readiness = {
	is_shown = {
		var:var_readiness < 5
		not = {
			has_variable = changed_readiness
		}
	}
}
